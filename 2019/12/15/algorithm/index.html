//加法 进位 和 当前位分别计算，再当成两个操作数继续相加
int plus(int a, int b) {
int aTemp = 0, bTemp = 0;

while (b != 0) {
    aTemp = a ^ b;
    bTemp = (a & b) << 1;
    a = aTemp;
    b = bTemp;
}
return a;

}
